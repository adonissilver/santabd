SELECT first_name, last_name, salary       -- 1ª parte: o que você quer ver (colunas)
FROM employees                              -- 2ª parte: de qual tabela os dados vêm
WHERE department_id = 10                    -- 3ª parte (opcional): filtro de quais linhas
ORDER BY salary DESC;                        -- 4ª parte (opcional): como ordenar os resultados

SELECT → diz quais colunas você quer mostrar.
FROM → diz de qual tabela os dados vêm.
WHERE → diz quais linhas incluir (condição).
ORDER BY → diz em que ordem mostrar os resultados.
